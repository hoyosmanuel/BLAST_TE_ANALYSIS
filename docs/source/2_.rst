  #!/bin/bash
  #SBATCH --job-name=fromGTF
  #SBATCH --output=%x.%A_%a.out
  #SBATCH --error=%x.%A_%a.err
  #SBATCH --partition=nocona
  #SBATCH --array=0-77
  #SBATCH --nodes=1
  #SBATCH --ntasks=1
  
  gtf=/lustre/scratch/mhoyosro/project3/RESULTS_RNA_MAPPING
  out=/lustre/scratch/mhoyosro/project3/TRANSCRIPTOME_SOLO
  
  mkdir -p "$out"
  
  mapfile -t gtfs < <(find "$gtf" -name "*_MM.gtf" | sort)
  
  input=${gtfs[$SLURM_ARRAY_TASK_ID]}
  
  species=$(basename "$(dirname "$input")")
  base=$(basename "$input" .gtf)
  
  mkdir -p "$out/$species"
  
  echo "INPUT GTF: $input"
  echo "SPECIES: $species"
  echo "OUTPUT: $out/$species/${base}_exones.gtf"
  
  awk '$3 == "exon"' "$input" > "$out/$species/${base}_exones.gtf"
