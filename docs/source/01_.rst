2) Get the RNAseq data
-----------------------
.. code-block:: bash

  nano getdata.sh

.. code-block:: bash

  #!/bin/bash
  #SBATCH --job-name=descarga
  #SBATCH --output=%x.%j.out
  #SBATCH --error=%x.%j.err
  #SBATCH --partition=nocona
  #SBATCH --nodes=8
  #SBATCH --ntasks=8
  
  cd /lustre/scratch/mhoyosro/project3/
  export PATH=/lustre/work/mhoyosro/software/sratoolkit/bin:${PATH}
  
  prefetch --max-size 99999999999 SRR18652231
  prefetch --max-size 99999999999 SRR18652230
  prefetch --max-size 99999999999 SRR18652219
  prefetch --max-size 99999999999 SRR18652218
  prefetch --max-size 99999999999 SRR18652217
  prefetch --max-size 99999999999 SRR18652128
  prefetch --max-size 99999999999 SRR18652132
  prefetch --max-size 99999999999 SRR18652131
  prefetch --max-size 99999999999 SRR18652130
  prefetch --max-size 99999999999 SRR18652129
  prefetch --max-size 99999999999 SRR18652220
  prefetch --max-size 99999999999 SRR18652180
  prefetch --max-size 99999999999 SRR18652229
  prefetch --max-size 99999999999 SRR18652228

.. Then unpack the data inside of this directory:
.. /lustre/scratch/mhoyosro/project3/RNAs

.. code-block:: bash

  nano unpack.sh

.. code-block:: bash

  #!/bin/bash
  #SBATCH --job-name=unpack
  #SBATCH --output=%x.%j.out
  #SBATCH --error=%x.%j.err
  #SBATCH --partition=nocona
  #SBATCH --nodes=8
  #SBATCH --ntasks=8
  
  cd /lustre/scratch/mhoyosro/project3/RNAs
  
  export PATH=/lustre/work/mhoyosro/software/sratoolkit/bin:${PATH}
  
  fastq-dump *.sra
