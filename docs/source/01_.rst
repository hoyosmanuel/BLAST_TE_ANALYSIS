2) Align the RNAseq data to each genome
---------------------------------------

.. code-block:: bash

  nano minimap2.sh

.. code-block:: bash

  #!/bin/bash
  #SBATCH --job-name=minimap_rna
  #SBATCH --output=logs/%x_%j.out
  #SBATCH --error=logs/%x_%j.err
  #SBATCH --partition=nocona
  #SBATCH --nodes=1
  #SBATCH --ntasks=36
  
  cd /lustre/scratch/mhoyosro/project3
  
  export PATH=/lustre/work/mhoyosro/software/minimap2:${PATH}
  
  RNA_DIR=/lustre/scratch/mhoyosro/project3/RNA_DATA
  REF_DIR=/lustre/scratch/mhoyosro/project3/REF_GENOMES
  OUT_DIR=/lustre/scratch/mhoyosro/project3/RESULTS_RNA_MAPPING
  
  mkdir -p "$OUT_DIR" logs
  
  map_one () {
      FASTQ=$1
      SPECIES=$2
      REF=$3
  
      RUN=$(basename "$FASTQ" .fastq)
      REFERENCE="$REF_DIR/$REF"
      SPECIES_OUT="$OUT_DIR/$SPECIES"
  
      mkdir -p "$SPECIES_OUT"
  
      echo "======================================"
      echo "RUN: $RUN"
      echo "SPECIES: $SPECIES"
      echo "FASTQ: $FASTQ"
      echo "REFERENCE: $REFERENCE"
      echo "======================================"
  
      if [ ! -f "$FASTQ" ]; then
          echo "FASTQ not found: $FASTQ"
          return
      fi
  
      if [ ! -f "$REFERENCE" ]; then
          echo "Reference not found: $REFERENCE"
          return
      fi
  
      minimap2 -t 36 -ax splice:hq -uf "$REFERENCE" "$FASTQ" | \
      samtools sort -@ 8 -o "$SPECIES_OUT/${RUN}_MM.sorted.bam"
  
      samtools index "$SPECIES_OUT/${RUN}_MM.sorted.bam"
  
      samtools flagstat "$SPECIES_OUT/${RUN}_MM.sorted.bam" > \
      "$SPECIES_OUT/${RUN}_MM.flagstat.txt"
  
      echo "Done: $RUN"
  }
  
  # Vespertilio murinus
  for fq in $RNA_DIR/ERR1375784{5..9}.fastq; do
      map_one "$fq" Vespertilio_murinus mVesMur1.1.pri.fa
  done
