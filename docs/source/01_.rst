3) Align the RNAseq data to each genome
---------------------------------------

.. code-block:: bash

  nano minimap2.sh

.. code-block:: bash

  #!/bin/bash
  #SBATCH --job-name=minimap
  #SBATCH --output=%x.%j.out
  #SBATCH --error=%x.%j.err
  #SBATCH --partition=nocona
  #SBATCH --nodes=1
  #SBATCH --ntasks=36
  
  cd /lustre/scratch/mhoyosro/project3/RESULTS
  export PATH=/lustre/work/mhoyosro/software/minimap2:${PATH}
  
  rna=/lustre/scratch/mhoyosro/project3/RNAs
  
  ref=/lustre/scratch/mhoyosro/project3/GENOMES
  
  myotis_reference="$ref/mMyoMyo1.6.pri.fa"
  molossus_reference="$ref/mMolMol1.2.pri.fa"
  rousettus_reference="$ref/mRouAeg1.4.pri.fa"
  pipistrellus_reference="$ref/mPipKuh1.2.pri.fa"
  rhinolophus_reference="$ref/mRhiFer1.5.pri.fa"
  phyllostomus_reference="$ref/mPhyDis1.3.pri.fa"
  hipposideros_reference="$ref/mHipLar1.2.pri.fa"
  
  minimap2 -ax splice $molossus_reference $rna/SRR18652231.fastq > molossus/SRR18652231_MM.sam
  minimap2 -ax splice $molossus_reference $rna/SRR18652230.fastq > molossus/SRR18652230_MM.sam
  minimap2 -ax splice $myotis_reference $rna/SRR18652217.fastq > myotis/SRR18652217_MM.sam
  minimap2 -ax splice $myotis_reference $rna/SRR18652218.fastq > myotis/SRR18652218_MM.sam
  minimap2 -ax splice $myotis_reference $rna/SRR18652219.fastq > myotis/SRR18652219_MM.sam
  minimap2 -ax splice $phyllostomus_reference $rna/SRR18652128.fastq > phyllostomus/SRR18652128_MM.sam
  minimap2 -ax splice $phyllostomus_reference $rna/SRR18652129.fastq > phyllostomus/SRR18652129_MM.sam
  minimap2 -ax splice $phyllostomus_reference $rna/SRR18652130.fastq > phyllostomus/SRR18652130_MM.sam
  minimap2 -ax splice $phyllostomus_reference $rna/SRR18652131.fastq > phyllostomus/SRR18652131_MM.sam
  minimap2 -ax splice $phyllostomus_reference $rna/SRR18652132.fastq > phyllostomus/SRR18652132_MM.sam
  minimap2 -ax splice $pipistrellus_reference $rna/SRR18652220.fastq > pipistrellus/SRR18652220_MM.sam
  minimap2 -ax splice $rhinolophus_reference $rna/SRR18652180.fastq > rhinolophus/SRR18652180_MM.sam
  minimap2 -ax splice $rousettus_reference $rna/SRR18652228.fastq > rousettus/SRR18652228_MM.sam
  minimap2 -ax splice $rousettus_reference $rna/SRR18652229.fastq > rousettus/SRR18652229_MM.sam
  minimap2 -ax splice $hipposideros_reference $rna/SRR17418361.fastq > hipposideros/SRR17418361_MM.sam

  fastq-dump *.sra



4) Bamming/Sorting/Indexing the RNAseq aligment
-----------------------------------------------

4.1 Bamming
~~~~~~~~~~~

.. code-block:: bash

  nano bamming.sh

.. code-block:: bash

  #!/bin/bash
  #SBATCH --job-name=bamming
  #SBATCH --output=%x.%A_%a.out
  #SBATCH --error=%x.%A_%a.err
  #SBATCH --partition=nocona
  #SBATCH --array=0-14
  #SBATCH --nodes=1
  #SBATCH --ntasks=1
  
  cd /lustre/scratch/mhoyosro/project3/RESULTS
  
  . /home/mhoyosro/conda/etc/profile.d/conda.sh
  conda activate alineador
  
  # Array de archivos SAM
  sams=(
  "molossus/SRR18652231_MM.sam"
  "molossus/SRR18652230_MM.sam"
  "myotis/SRR18652217_MM.sam"
  "myotis/SRR18652218_MM.sam"
  "myotis/SRR18652219_MM.sam"
  "phyllostomus/SRR18652128_MM.sam"
  "phyllostomus/SRR18652129_MM.sam"
  "phyllostomus/SRR18652130_MM.sam"
  "phyllostomus/SRR18652131_MM.sam"
  "phyllostomus/SRR18652132_MM.sam"
  "pipistrellus/SRR18652220_MM.sam"
  "rhinolophus/SRR18652180_MM.sam"
  "rousettus/SRR18652228_MM.sam"
  "rousettus/SRR18652229_MM.sam"
  "hipposideros/SRR17418361_MM.sam"
  )
  
  sam=${sams[$SLURM_ARRAY_TASK_ID]}
  bam=${sam%.sam}.bam
  
  samtools view -bS "$sam" > "$bam"

.. code-block:: bash

  #!/bin/bash
  #SBATCH --job-name=gtfing
  #SBATCH --output=%x.%j.out
  #SBATCH --error=%x.%j.err
  #SBATCH --partition=nocona
  #SBATCH --array=0-14
  #SBATCH --nodes=1
  #SBATCH --ntasks=1
  
  cd /lustre/scratch/mhoyosro/project3/RESULTS
  export PATH=/lustre/work/mhoyosro/software/stringtie:${PATH}
  
  FILES=(
  "molossus/SRR18652231_MM"
  "molossus/SRR18652230_MM"
  "myotis/SRR18652217_MM"
  "myotis/SRR18652218_MM"
  "myotis/SRR18652219_MM"
  "phyllostomus/SRR18652128_MM"
  "phyllostomus/SRR18652129_MM"
  "phyllostomus/SRR18652130_MM"
  "phyllostomus/SRR18652131_MM"
  "phyllostomus/SRR18652132_MM"
  "pipistrellus/SRR18652220_MM"
  "rhinolophus/SRR18652180_MM"
  "rousettus/SRR18652228_MM"
  "rousettus/SRR18652229_MM"
  "hipposideros/SRR17418361_MM"
  )
  
  # Selec the file according to the index of the array
  BASE=${FILES[$SLURM_ARRAY_TASK_ID]}
  
  # StringTie excution
  stringtie -o ${BASE}.gtf ${BASE}.sorted.bam
