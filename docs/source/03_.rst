4. Clean the gff files
----------------------

.. code-block:: bash

  nano cleaning.sh

.. code-block:: bash

  #!/bin/bash
  #SBATCH --dependency=afterok:22343489
  #SBATCH --job-name=cleaner
  #SBATCH --output=%x.%j.out
  #SBATCH --error=%x.%j.err
  #SBATCH --partition=nocona
  #SBATCH --nodes=1
  #SBATCH --ntasks=16
  #SBATCH --mem=128G
  #SBATCH --time=48:00:00
  
  
  base="/lustre/scratch/mhoyosro/project3/PLANCE"
  
  declare -A species
  species[hipposideros]="TEs_in_hLar.60percent.gff"
  species[molossus]="TEs_in_mMol.60percent.gff"
  species[myotis]="TEs_in_mMyo.60percent.gff"
  species[phyllostomus]="TEs_in_pDis.60percent.gff"
  species[pipistrellus]="TEs_in_pKuh.60percent.gff"
  species[rhinolophus]="TEs_in_rFer.60percent.gff"
  species[rousettus]="TEs_in_rAeg.60percent.gff"
  
  for sp in "${!species[@]}"; do
      echo "Procesando $sp ..."
      cd "$base/$sp"
  
      for f in ${species[$sp]}; do
          echo "  limpiando $f ..."
          grep -v "^##PAF" "$f" > "${f%.gff}_clean.gff"
      done
  done
  
  echo "Listo."
